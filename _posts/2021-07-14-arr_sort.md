```
var quickSort = function(arr){
	if(arr.length < 1){
		return arr;
	}
	var centerIndex = Math.floor(arr.length / 2);
	var centerValue = arr[centerIndex];
	var left = [] , right = [];
    for( var i = 0; i < arr.lenght; i ++){
        if(arr[i] < centerValue){
            left.push(arr[i]);
        } else{
            right.push(arr[i]);
        }
    }
	return quickSort(left).concat([centerValue],quckSort(right));
}
```

## 冒泡排序 ##
![冒泡排序图](/img/sort/bubble-sort.gif)
```
var bubbleSort = function( arr) {
    var len = arr.length;
    for( var i = 0; i < len; i ++){
        for( var j = 0; j < len - 1 - i; j ++){
            if(arr[j] > arr[j + 1]) { //相邻元素两两对比
                var temp = arr[j + 1]; //元素交换
                arr[j + 1] = arr[j];
                arr[j] = temp;
            }
        }
    }
    return arr;
}


```
## 插入排序 ##
![插入排序图](/img/sort/insert-sort.gif)

```
var insertSort = function( arr) {
    var len = arr.length;
    var preIndex, current;
    for( var i = 1; i < len; i ++){
        preIndex = i - 1;
        current = arr[i];
        while(preIndex >= 0 && arr[preIndex] > current){
            arr[preIndex + 1] = arr[preIndex];
            preIndex --;
        }
        arr[preIndex + 1] = current;
    }
    return arr;
}

```

```
for(var i=0;i<arr.length-1;i++){       //大循环，需要比对的次数
    var min = arr[i];                  //假定一个最小值
```

## js sort方法排序 ##

```
知识点：sort()不传参数默认是升序，且按照**字符串**比较方式排序;且会**改变原数组**
//升序
arr.sort(function(a,b){
	return a - b;
})
//降序
arr.sort(function(a,b){
	return b - a;
})

//根据数组对象属性排序

arr.sort(function(a,b){
	return a.property - b.property;
})
```
